-- 1. 시퀀스 생성
CREATE SEQUENCE seq_order_id
START WITH 1
INCREMENT BY 1
NOCACHE
NOCYCLE;

--  INSERT 시 시퀀스 사용
INSERT INTO orders (order_id, customer_id, order_date)
VALUES (seq_order_id.NEXTVAL, 'user01', SYSDATE);
장점: Oracle 10g 이전 버전에서도 사용 가능

--시퀀스 조회하기
select   seq_order_id.nextval  from dual;

단점: INSERT 시 항상 NEXTVAL 명시해야 함

 
 자동생성 (오라클12버전 이상)
CREATE TABLE orders (
    order_id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    customer_id VARCHAR2(50),
    order_date DATE DEFAULT SYSDATE
);


GENERATED BY DEFAULT AS IDENTITY → 자동 증가
INSERT 시 order_id를 생략 가능

 INSERT INTO orders (customer_id, order_date)
VALUES ('user01', SYSDATE);
자동으로 order_id가 증가해서 들어감

select  name,id ,   level,  lpad( '  ' ,  5 *( level -1) )  ||  name    levelname
from acorn
start  with     manager_id is  null
connect by prior  id  =  manager_id
order by level;



SELECT name,
       id,
       LEVEL,
       LPAD(' ', 5 * (LEVEL - 1)) || name AS levelname
FROM acorn
START WITH manager_id IS NULL
CONNECT BY PRIOR id = manager_id
ORDER SIBLINGS BY id;



select name, id , sys_connect_by_path(name , '->')
from acorn
start with  manager_id is null
connect by prior  id  =  manager_id
order siblings by  id;
