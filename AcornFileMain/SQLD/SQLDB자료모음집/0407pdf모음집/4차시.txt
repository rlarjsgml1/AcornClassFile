 ########
 sum() over() 사용하여 누적합 구하기
 ########

--전체 누적값 구하기
 SELECT  p_total,  sum(p_total) over(  )
 FROM panmae;


--첫행에서 현재행까지의 누적 구하기 ,  p_total 컬럼 기준으로
 SELECT  p_total,  sum(p_total) over( order by p_total)  
 FROM panmae;

--
 SELECT  p_total,  sum(p_total) over(
    order by p_total
    rows between unbounded preceding and current row 
   )
 FROM panmae;


--  unbounded preceding  (제한없는 이전의 뜻 : 처음부터라고 해석하면 됨)

 --첫행에서 현재행까지의 누적 구하기
 --기본값은 아래와 같다, 생략할 수 있다
 --같은 값은  같은  range로 묶인다
--필요하면  rows로 변경해야 한다
 SELECT  p_total,  sum(p_total) over( order by p_total
    range between unbounded preceding and current row   )
 FROM panmae;

###########
   pivot  형식
##########

SELECT *
FROM (
    -- 1. 원본 데이터
    SELECT 컬럼1, 컬럼2, 값컬럼
    FROM 테이블
)
PIVOT (
    -- 2. 집계 함수
    집계함수(값컬럼)
    
    -- 3. 기준 컬럼 (열로 바뀔 대상)
    FOR 기준컬럼
    
    -- 4. 실제 변환될 값들
    IN ('값1', '값2', '값3')
);
