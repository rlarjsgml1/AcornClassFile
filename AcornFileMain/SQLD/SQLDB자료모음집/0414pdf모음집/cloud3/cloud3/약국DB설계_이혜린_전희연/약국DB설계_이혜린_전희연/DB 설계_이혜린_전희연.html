🔗ERD

⚙️ 관계 구조

✅ 환자 ← 처방전 → 약국
  • 환자가 어느 약국에서 처방전을 받았는지 연결
    • 환자(1) : 처방전(N) → 1:N
    • 약국(1) : 처방전(N) → 1:N

✅ 처방전 ← 처방전_약품 → 약품

  • 처방전에 어떤 약품이 포함되는지 연결
  • 처방전과 약품 테이블 사이의 N:N 관계 해소하기 위한 중간 테이블
    • 처방전(1) : 처방전_약품(N) → 1:N
    • 약품(1) : 처방전_약품(N) → 1:N

⭐ 구현

테이블 생성 ( CREATE )

--1. 환자 테이블 ( 환자번호, 이름, 생년월일, 연락처, 성별, 주소 )
CREATE TABLE PEOPLE (   
    people_id      VARCHAR2(10) PRIMARY KEY,        
    people_name    VARCHAR2(50) NOT NULL, 
    birthday       DATE,
    phone          VARCHAR2(20) UNIQUE,
    gender         CHAR(1) CHECK (gender IN ('M','F')),
    address        VARCHAR2(100)
);
--2. 약품 테이블 ( 약품번호, 약품명, 제조사, 악품분류, 가격 )
CREATE TABLE MEDICINE ( 
    medicine_id      VARCHAR2(10) PRIMARY KEY,       
    medicine_name    VARCHAR2(100) NOT NULL,
    maker            VARCHAR2(30),
    category         VARCHAR2(30),
    price            NUMBER(10,2) CHECK (price >= 0)
);
--3. 약국 테이블 ( 약국번호, 약국이름, 주소, 전화번호 )
CREATE TABLE MEDICINE_STORE (    
    store_id        VARCHAR2(10) PRIMARY KEY,
    store_name      VARCHAR2(15),   
    store_addr      VARCHAR2(100),
    store_phone     VARCHAR2(20) UNIQUE
);
--4. 처방전 테이블 ( 처방전고유번호, 환자번호, 약국번호, 발급일자, 질병명, 비고 )
CREATE TABLE RECIPE (   
    recipe_id        VARCHAR2(10) PRIMARY KEY,         
    issue_date       DATE NOT NULL,
    people_id        VARCHAR2(10) NOT NULL,
    store_id         VARCHAR2(10) NOT NULL,
    note             VARCHAR2(50),
    CONSTRAINT fk_rec_people
        FOREIGN KEY (people_id) REFERENCES PEOPLE(people_id),
    CONSTRAINT fk_rec_store
        FOREIGN KEY (store_id) REFERENCES MEDICINE_STORE(store_id)
);
--5. 처방전_약품 테이블 ( 처방전고유번호, 약품번호, 1화투여량, 복용횟수, 투약일수 )
CREATE TABLE RECIPE_MED (    
    recipe_id         VARCHAR2(10) NOT NULL,
    medicine_id       VARCHAR2(10) NOT NULL,
    dosage            NUMBER CHECK (dosage > 0),      
    times_per_day     NUMBER CHECK (times_per_day > 0),  
    total_days        NUMBER CHECK (total_days > 0),   
    CONSTRAINT pk_rec_med PRIMARY KEY (recipe_id, medicine_id),
    CONSTRAINT fk_pd_rec FOREIGN KEY (recipe_id)
																       REFERENCES RECIPE(recipe_id),
    CONSTRAINT fk_pd_med FOREIGN KEY (medicine_id)
															         REFERENCES MEDICINE(medicine_id)
);

데이터 조회 ( SELECT )

환자 정보 조회

SELECT people_name "환자이름", birthday "생년월일",
			TRUNC(MONTHS_BETWEEN(SYSDATE, birthday)/12) || '세' AS "나이",
			phone "연락처",
			DECODE(gender, 'M', '남', 'F', '여', '-') "성별",
			address "주소"
FROM PEOPLE;

약국 정보 조회

SELECT 
    store_name AS 약국명,
    store_phone AS 연락처,
    store_addr AS 주소,
    CASE SUBSTR(store_phone, 1, INSTR(store_phone, '-') - 1)
        WHEN '02'  THEN '서울'
        WHEN '031' THEN '경기도'
        WHEN '051' THEN '부산'
        ELSE '기타'
    END AS 지역_분류
FROM MEDICINE_STORE;

약품별 평균 복용일수와 최대 복용일수 조회

SELECT 
    m.medicine_name 약품명,
    ROUND(AVG(rm.total_days), 2) || '일' AS 평균_복용일수,
    MAX(rm.total_days) || '일' AS 최대_복용일수
FROM MEDICINE m
JOIN RECIPE_MED rm ON m.medicine_id = rm.medicine_id
GROUP BY m.medicine_name
ORDER BY 평균_복용일수 DESC;

FROM RECIPE_med  rm
JOIN MEDICINE m
ON rm.medicine_id = m.medicine_id
JOIN RECIPE r
ON rm.recipe_id = r.recipe_id;
--3. 1개의 처방전별 총 금액 (최종)
SELECT rm.recipe_id 처방전고유번호, 
				TO_CHAR(SUM(m.price), '999,999') || '원' 가격

ON rm.medicine_id = m.medicine_id

ON rm.recipe_id = r.recipe_id

약품별 매출 TOP5 조회

🔎 상세보기 

--1. 조인 (처방전, 처방전_약품, 약품)
SELECT *
FROM RECIPE_med  rm
JOIN MEDICINE m
ON rm.medicine_id = m.medicine_id
JOIN RECIPE r
ON rm.recipe_id = r.recipe_id;

--2. 원하는 컬럼 조회
SELECT medicine_name  약품명, price 가격
FROM RECIPE_med  rm
JOIN MEDICINE m
ON rm.medicine_id = m.medicine_id
JOIN RECIPE r
ON rm.recipe_id = r.recipe_id;
 
--3. 금액별 순위 매기기 (FROM절 서브쿼리)
SELECT     medicine_name AS 약품명,
           SUM(price) AS 총합,
           ROW_NUMBER() OVER (ORDER BY SUM(price) DESC) AS rn
FROM   RECIPE_med rm
JOIN   MEDICINE m 
ON rm.medicine_id = m.medicine_id
JOIN   RECIPE r   
ON rm.recipe_id   = r.recipe_id
GROUP  BY medicine_name;
--4. 약품 매출 TOP5 조회
SELECT 약품명, 총합
FROM (
    SELECT medicine_name AS 약품명,
           SUM(price) AS 총합,
           ROW_NUMBER() OVER (ORDER BY SUM(price) DESC) AS rn
    FROM   RECIPE_med rm
    JOIN   MEDICINE m ON rm.medicine_id = m.medicine_id
    JOIN   RECIPE r   ON rm.recipe_id   = r.recipe_id
    GROUP  BY medicine_name
)
WHERE rn <= 5
ORDER BY 총합 DESC;

지역별 약국 목록 조회

🔎 상세보기

--지역번호 추출 -> 지역 표시 ( DECODE 활용 )
SELECT store_name, store_phone, store_addr, 
				SUBSTR(store_phone, 1, INSTR(store_phone, '-')-1) 지역번호,
        DECODE(SUBSTR(store_phone, 1, INSTR(store_phone, '-')-1), 
        '02', '서울', '031', '경기도', '051', '부산') 지역
FROM MEDICINE_STORE;


--FROM절 서브쿼리
SELECT DECODE(SUBSTR(store_phone, 1, INSTR(store_phone, '-')-1), 
				'02', '서울', '031', '경기도', '051', '부산') 지역
FROM medicine_store;


--FROM절 서브쿼리 필요성
SELECT DECODE(SUBSTR(store_phone, 1, INSTR(store_phone, '-')-1), 
								'02', '서울', '031', '경기도', '051', '부산') 지역
			,store_name
FROM MEDICINE_STORE
WHERE DECODE(SUBSTR(store_phone, 1, INSTR(store_phone, '-')-1), 
			'02', '서울', '031', '경기도', '051', '부산') = '서울';
--서울 지역에 위치한 약국명
SELECT 지역, store_name 약국명
FROM (
    SELECT DECODE(
             SUBSTR(store_phone, 1, INSTR(store_phone, '-') - 1),
             '02',  '서울',
             '031', '경기도',
             '051', '부산'
           ) AS 지역,
           store_name
    FROM MEDICINE_STORE
)
WHERE 지역 = '서울';
